迭代迴圈 (Repeating events)

用 for 迴圈數數字

            
            # range() 函式產生 4 個連續整數
            # 從 0 開始
            for steps in range(4):
                print(steps)
            
        
            
            # 如果想要從 1 開始
            # range() 函式裡面放兩個參數
            # 第 1 個表示起始值，第 2 個表示中止值
            for steps in range(1, 4):
                 print(steps)
            
        
            
            # range() 函式第 3 個參數表示間隔值
            for steps in range(1, 10, 2):
                print(steps)

星號三角

            
            for i in range(1, 6):
                print('*'*i)

用 for 迴圈讀取一堆數值

            
            # list 容器：使用中括號包含一堆資料
            for steps in [1, 2, 3, 4, 5]:
                print(steps)
            
        
            
            for steps in [1, 3, 5, 2, 4]:
                print(steps)

用 for 迴圈讀取一堆字串

            
            for color in ['red', 'blue', 'green', 'black']:
                print(color)

for 迴圈執行次數是固定的

            
            for steps in range(1, 21):
                print('Name Card', steps)

條件迴圈 (Repeating events until done)

while 迴圈

            
            # 只要執行條件維持成立，while 迴圈就會一直執行
            # 直到條件轉變為不成立，才會中止並離開迴圈
            # 執行條件反過來，可以視為離開迴圈的中止條件
            answer = "0"
            while answer != "4":
                answer = input("What is 2 + 2? ")
            print("Yes! 2 + 2 = 4")

無條件進入迴圈

            
            counter = 1
            while True:
                print(counter)
                if counter == 10:
                    break
                else:
                    counter += 1

用布林變數控制迴圈

            
            plus_one = True
            int_sum = 0
            while plus_one:
                int_sum = int_sum + 1
                print(int_sum)
                if int_sum == 10:
                    plus_one = False

無窮迴圈

            
            # 兩行程式就會造成無窮迴圈
            # 同時按下 [Ctrl] + [C] 中斷無窮迴圈
            while True:
                print('infinite')

兩種迴圈比較

            
            print('for loop')
            counter = 0
            for i in range(10):
                counter += 1
                print(counter)

            print('while loop')
            counter = 0
            while counter != 10:
                counter += 1
                print(counter)

陣列操作 (Remembering lists)

用 list 儲存字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests)

用 list 儲存數值資料

            
            scores = [78, 85, 62, 49, 98]
            print(scores)

以索引編號從 list 中取出 1 個字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests[0])

以索引編號從 list 中取出 1 個數值資料

            
            scores = [78, 85, 62, 49, 98]
            print(scores[3])

從 list 中倒數取出字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests[-1])
            print(guests[-2])

改變 list 中的字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print("first value is " + guests[0])
            guests[0] = 'Steve'
            print("first value is now " + guests[0])

使用 list.append() 函式新增字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests)
            guests.append('Steve')
            print(guests)
            print(guests[-1])

使用 list.remove() 函式移除字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests)
            guests.remove('Christopher')
            print(guests)
            print(guests[0])

使用 del 指令刪除字串資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests)
            del guests[0]
            print(guests)
            print(guests[0])

回傳指定字串資料的索引編號

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests.index('Bill'))
            
        
            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests.index('Steve'))

用 for 迴圈逐一讀取 list 資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            for steps in range(4):
                print(guests[steps])

用 len() 函式計算 list 資料個數

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            nbrEntries = len(guests)
            print(nbrEntries)
            print()
            for steps in range(nbrEntries):
                print(guests[steps])

用 for 迴圈逐一讀取 list 資料儲存到迭代變數

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            for guest in guests:
                print(guest)

用 list.sort() 函式排序 list 資料

            
            guests = ['Christopher', 'Susan', 'Bill', 'Satya']
            print(guests)
            print()
            guests.sort()
            print(guests)
            for guest in guests:
                print(guest)

在迴圈中持續新增資料到 list

            
            guests = []
            name = " "
            while name != "DONE":
                name = input("Enter guest name (enter DONE if no more names) : ")
                guests.append(name)
            guests.sort()
            for guest in guests:
                print(guest)
            
        
            
            guests = []
            name = " "
            while name != "DONE":
                name = input("Enter guest name (enter DONE if no more names) : ")
                if name.upper() != "DONE":
                    guests.append(name)
            guests.sort()
            for guest in guests:
                print(guest)

存入檔案 (How to save information in files)

存取模式

            
            # r 讀取檔案
            # w 寫入檔案
            # a 增加檔案的內容

開啟一個檔案準備寫入資料

            
            fileName = "GuestList.txt"
            accessMode = "w"
            myFile = open(fileName, accessMode)

開啟檔案後寫入資料

            
            fileName = "GuestList.txt"
            accessMode = "w"
            myFile = open(fileName, accessMode)
            myFile.write("Hi there! ")
            myFile.write("How are you?")
            
        
            
            fileName = "GuestList.txt"
            accessMode = "w"
            myFile = open(fileName, accessMode)
            myFile.write("Hi there!\n")
            myFile.write("How are you?")
            # 關閉檔案
            myFile.close()

來賓資料建檔

            
            fileName = "GuestList.csv"
            accessMode = "w"
            myFile = open(fileName, accessMode)
            while True:
                new_guest = input('Continue or quit >')
                if new_guest == 'quit':
                    break
                first_name = input('First name:').title()
                last_name = input('Last name:').title()
                gender = input('Gender:').title()
                position = input('Position:')
                row = ','.join((first_name, last_name, gender, position)) + '\n'
                myFile.write(row)
            myFile.close()

讀取檔案 (Reading from files)

讀取來賓資料

            
            import csv
            fileName = "GuestList.csv"
            accessMode = "r"
            with open(fileName, accessMode) as myCSVFile:
                dataFromFile = csv.reader(myCSVFile)
                for row in dataFromFile:
                    print(row)
            
        
            
            import csv
            fileName = "GuestList.csv"
            accessMode = "r"
            with open(fileName, accessMode) as myCSVFile:
                dataFromFile = csv.reader(myCSVFile)
                for row in dataFromFile:
                    print(row)
                    for value in row:
                        print(value)

函式設計 (Functions)

設計函式

            
            def printMessage():
                print('Hello World')
                return

呼叫函式

            
            def printMessage():
                print('Hello World')
                return

            printMessage()

設定函式參數

            
            def printMessage(message):
                print(message)
                return

            printMessage('Hello world!')
            
        
            
            def displayMessage(greeting, name):
                message = greeting + ', ' + name
                print(message)
                return

            displayMessage('Hi', 'Christopher')

回傳資料

            
            def getMessage(name):
                message = 'Hello, ' + name
                return message

            def printMessage(message):
                print(message)
                return

            output = getMessage('Christopher')
            printMessage(output)

            try:
                result = firstNumber / secondNumber

            except:
                print("I am sorry something went wrong")

顯示錯誤類別

            
            import sys
            first = input("Enter the first number: ")
            second = input("Enter the second number: ")
            firstNumber = float(first)
            secondNumber = float(second)
            try:
                result = firstNumber / secondNumber
                print(first + " / " + second + " = " + str(result))
            except:
                error = sys.exc_info()[0]
                print("I am sorry something went wrong")
                print(error)

處理特定錯誤類別

            
            import sys
            first = input("Enter the first number: ")
            second = input("Enter the second number: ")
            firstNumber = float(first)
            secondNumber = float(second)
            try:
                result = firstNumber / secondNumber
                print (first + " / " + second + " = " + str(result))
            except ZeroDivisionError:
                print("The answer is infinity")
            except:
                error = sys.exc_info()[0]
                print("I am sorry something went wrong")
                print(error)
