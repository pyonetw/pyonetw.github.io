字串變數 (String variables)

要求使用者輸入資訊

            
            # 要在 input() 函式中放入提示訊息
            # 使用者會依提示訊息輸入相關資訊
            # 無論使用者輸入什麼都是字串
            # 用 1 個變數來記住使用者輸入的內容
            name = input("What is your name? ")
            print(name)
            
        
            
            # 用 input() 函式讓使用者輸入名字
            # 會將輸入的名字存到變數 name
            # 存的時候要用 = 號
            name = input("What is your name? ")

            # 取的時候直接使用變數名稱
            # 將 name 放到 print() 函式中處理輸出
            print(name)
            
        
            
            # 使用者輸入什麼就存什麼
            name = input("What is your name? ")
            # 輸出前面輸入的名字
            print(name)
            # 改為 Mary
            name = "Mary"
            # 再輸出一次就變成是 Mary
            print(name)

調整字串變數資料顯示格式

            
            message = 'Hello world'
            # 全部變小寫
            print(message.lower())
            # 全部變大寫
            print(message.upper())
            # 大寫變小寫，小寫變大寫
            print(message.swapcase())
            # 注意！因為沒有設 message = message.swapcase()
            # 所以 message 真正的內容還是沒變
            print(message)

其他字串函式

            
            message = 'Hello world'
            # 在 message 中找 'world'，找得到就回傳第一個字元的 index (位置)
            # index 從 0 開始，'H' 的 index 就是 0
            # 找到 'world'，回傳 'w' 的 index 就是 6
            # 找不到會回傳 -1
            print(message.find('world'))
            # 計算字串中有幾個 'o'
            print(message.count('o'))
            # 第一個單字的首字母大寫
            print(message.capitalize())
            # 將 'Hello' 取代為 'Hi'
            print(message.replace('Hello', 'Hi'))

格式化顯示數值

            
            # 以整數表示
            print('I have %d cats' % 6)
            # 以 3 個字元寬度顯示整數，不滿 3 個則補空白
            print('I have %3d cats' % 6)
            # 以 3 個字元寬度顯示整數，不滿 3 個則補 0
            print('I have %03d cats' % 6)
            # 以浮點數表示，預設小數位數為 6 位
            print('I have %f cats' % 6)
            # 以浮點數表示，設定小數位數為 2 位
            print('I have %.2f cats' % 6)

格式化顯示數值

            
            # 以整數表示
            print("I have {0:d} cats".format(6))
            # 以 3 個字元寬度顯示整數，不滿 3 個則補空白
            print("I have {0:3d} cats".format(6))
            # 以 3 個字元寬度顯示整數，不滿 3 個則補 0
            print("I have {0:03d} cats".format(6))
            # 以浮點數表示，預設小數位數為 6 位
            print("I have {0:f} cats".format(6))
            # 以浮點數表示，設定小數位數為 2 位
            print("I have {0:.2f} cats".format(6))

日期時間 (Working with dates and times)

取得今天日期

            
            import datetime
            print(datetime.date.today())

將日期存到變數

            
            import datetime

            import datetime
            currentDate = datetime.date.today()
            print(currentDate)
            print(currentDate.year)
            print(currentDate.month)
            print(currentDate.day)

使用 strftime() 函式設定格式化日期

            
            # http://strftime.org/
            # %d 就是日期
            # %B 月份正寫
            # %b 月份的縮寫
            # %Y 以 4 位數表示西元年份
            # %y 以 2 位數表示西元年份
            # %A 星期正寫
            # %a 星期縮寫
            import datetime
            currentDate = datetime.date.today()
            print (currentDate.strftime('%d %b,%Y'))

            import datetime
            birthday = input("What is your birthday? ")
            birthdate = datetime.datetime.strptime(birthday,"%m/%d/%Y").date()

計算距離下次生日的天數

            
            import datetime
            nextBirthday = datetime.datetime.strptime('8/7/2019','%m/%d/%Y').date()

簡單決策 (Making decisions with code)

使用 if 決策式

            
            # 條件成立才執行相關程式
            answer = input("Would you like express shipping? ")
            if answer == "yes":
                print("That will be an extra $10")

關係運算

            
            ==  相等
            !=  不等
            <   小於
            >   大於
            <=  小於等於
            >=  大於等於

與條件相關或不相關

            
            # 條件成立會執行相關程式
            # 相關程式必須在 if 決策式條件下方縮排
            # 沒有縮排表示和條件無關
            answer = input("Would you like express shipping? ")
            if answer == "yes":
                print("That will be an extra $10")
            print("Have a nice day")

注意條件式比較內容

            
            answer = input("Would you like express shipping? ")
            # 將變數 answer 內容轉成小寫再判斷
            if answer.lower() == "yes":
                print("That will be an extra $10")
            print("Have a nice day")

條件式使用數值資料比較

            
            deposit = 150
            if deposit > 100:
                print("You get a free toaster!")
            print("Have a nice day")

使用輸入資料比較

            
            deposit = input("How much would you like to deposit? ")
            if deposit > 100:
                print("You get a free toaster!")
            print("Have a nice day")

數值比較要將字串轉換成數值

            
            # 方法一：輸入後先存到變數再轉換
            deposit = input("How much would you like to deposit? ")
            if int(deposit) > 100:
                print("You get a free toaster!")
            print("Have a nice day")
            
        
            
            # 方法二：輸入後先轉換再存到變數
            deposit = int(input("How much would you like to deposit? "))
            if deposit > 100:
                print("You get a free toaster!")
            print("Have a nice day")

條件分支

            
            # 條件不成立，才會執行 else 裡面程式
            deposit = input("How much would you like to deposit? ")
            if float(deposit) > 100:
                print("You get a free toaster!")
            else:
                print("Enjoy your mug!")
            print("Have a nice day")

使用布林變數儲存條件比較結果

            
            deposit = input("how much would you like to deposit? ")
            if float(deposit) > 100:
                freeToaster = True
            if freeToaster:
                print("enjoy your toaster")

確保變數先設定且有預設值

            
            freeToaster = False
            deposit= input("how much would you like to deposit? ")
            if float(deposit) > 100:
                freeToaster = True
            if freeToaster:
                print("enjoy your toaster")

錯誤修正

            
            deposit = input("How much would you like to deposit? ")
            if deposit > 100
                print("You get a free toaster!")
            freeToaster = true
            else:
                print("Enjoy your mug!")
            print("Have a nice day")

複雜決策 (Complex decisions with code)

條件分支

            
            # 一個條件可能會因為不同情況，導致不同結果
            country = input("Where are you from? " )
            # 在 if 決策式中使用 elif 加以比較其他情況
            if country == "CANADA":
                print("Hello")
            elif country == "GERMANY":
                print("Guten Tag")
            elif country == "FRANCE":
                print("Bonjour")
            
        
            
            country = input("Where are you from? " )
            if country == "CANADA":
                print("Hello")
            elif country == "GERMANY":
                print("Guten Tag")
            elif country == "FRANCE":
                print("Bonjour")
            else:
                print("Aloha/Ciao/G’Day")

and 邏輯

            
            # 使用 and 運算，必須所有條件成立才算成立
            wonLottery = True
            bigWin = True
            if wonLottery and bigWin:
                print("you can retire")

or 邏輯

            
            # 使用 or 運算，任何一個條件成立就算成立
            saturday = True
            sunday = False
            if saturday or sunday:
                print("you can sleep in")

複合條件：一種運算

            
            # 一個 if 敘述式中，可以同時使用兩個以上的 and 或 or 運算
            month = "Jun"
            if month == "Sep" or month =="Apr" \
                or month == "Jun" or month == "Nov":
                print("There are 30 days in this month")
            
        
            
            favMovie = "Star Wars"
            favBook = "Lord of the Rings"
            favEvent = "ComiCon"

            if favMovie == "Star Wars" \
                and favBook == "Lord of the Rings" \
                and favEvent == "ComiCon":
                print("You and I should hang out")

複合條件：兩種運算

            
            # 一個 if 敘述式中，可以同時混用兩個以上的 and 和 or 運算
            country = "CANADA"
            # country = "VIETNAM"
            pet = "MOOSE"
            # pet = "BEAVER"

            if country == "CANADA" and \
                pet == "MOOSE" or pet == "BEAVER":
                print("Do you play hockey too?")
            
        
            
            country = "CANADA"
            # country = "VIETNAM"
            pet = "MOOSE"
            # pet = "BEAVER"

            if country == "CANADA" and \
                (pet == "MOOSE" or pet == "BEAVER"):
                print("Do you play hockey too?")
