Turtle 海龜繪圖程式

一條直線

            
            import turtle
            window = turtle.Screen()
            turtle.forward(100)
            window.exitonclick()

三條不同色線

            
            import turtle
            window = turtle.Screen()
            turtle.color('green')
            turtle.forward(100)
            turtle.right(45)
            turtle.color('blue')
            turtle.forward(50)
            turtle.right(45)
            turtle.color('pink')
            turtle.forward(100)
            window.exitonclick()

正方形

            
            import turtle

            window = turtle.Screen()

            for step in range(4):
                turtle.forward(100)
                turtle.right(90)

            window.exitonclick()

田方形

            
            import turtle

            window = turtle.Screen()

            for steps in range(4):
                turtle.forward(100)
                turtle.right(90)
                for moresteps in range(4):
                    turtle.forward(50)
                    turtle.right(90)

            window.exitonclick()

星星

            
            import turtle

            star = turtle.Turtle()

            for i in range(5):
                star.forward(100)
                star.right(144)

            turtle.done()

五角星星

            
            import turtle

            window = turtle.Screen()

            for steps in range(5):
                turtle.forward(100)
                turtle.right(360/5)
                for moresteps in range(5):
                    turtle.forward(50)
                    turtle.right(360/5)

            window.exitonclick()

多邊形

            
            import turtle

            polygon = turtle.Turtle()
            num_sides = 6
            side_length = 70
            angle = 360.0 / num_sides

            for i in range(num_sides):
                polygon.forward(side_length)
                polygon.right(angle)

            turtle.done()

六角星星

            
            import turtle

            window = turtle.Screen()
            num_sides = 6

            for steps in range(num_sides):
                turtle.forward(100)
                turtle.right(360/num_sides)
                for moresteps in range(num_sides):
                    turtle.forward(50)
                    turtle.right(360/num_sides)

            window.exitonclick()

圓形

            
            import turtle

            window = turtle.Screen()
            turtle.speed(10)
            circles = 6
            angle = 360/circles

            for i in range(circles):
                turtle.circle(50)
                turtle.penup()
                turtle.forward(100)
                turtle.right(angle)
                turtle.pendown()

            window.exitonclick()

點網格

            
            import turtle
            from random import choice

            dot_grid = turtle.Turtle()
            dot_distance = 25
            width = 6
            height = 6
            colors = ('red', 'green', 'blue')

            dot_grid.penup()
            for y in range(height):
                for i in range(width):
                    dot_grid.color(choice(colors))
                    dot_grid.dot()
                    dot_grid.forward(dot_distance)
                dot_grid.backward(dot_distance * width)
                dot_grid.right(90)
                dot_grid.forward(dot_distance)
                dot_grid.left(90)

            turtle.done()
